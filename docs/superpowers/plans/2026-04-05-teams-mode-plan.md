# Teams Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add persistent Teams with Continuous Flow workflow, real-time presence, and segment assignment to CATforCAT.

**Architecture:** New Team/TeamMember entities with reusable membership. Segments get a workflowStage field for pipeline tracking. Real-time presence via heartbeat polling. PM assigns segment ranges; permissions enforced by range, not locks. 8 team colors as CSS variables across 6 themes.

**Tech Stack:** Next.js 16 (App Router), Prisma 6, PostgreSQL 18, TypeScript strict, Tailwind v4, Zustand, CSS variables.

**Spec:** `docs/superpowers/specs/2026-04-05-teams-mode-design.md`

model Team {
  id          String   @id @default(uuid())
  name        String
  description String?
  ownerId     String   @map("owner_id")
  createdAt   DateTime @default(now()) @map("created_at")

  owner    User         @relation("TeamOwner", fields: [ownerId], references: [id], onDelete: Cascade)
  members  TeamMember[]
  projects Project[]    @relation("TeamProjects")

  @@map("teams")
}

model TeamMember {
  id       String   @id @default(uuid())
  teamId   String   @map("team_id")
  userId   String   @map("user_id")
  role     String   // pm | translator | reviewer | proofreader | terminologist | dtp
  color    String   @default("azul") // rojo | rosa | morado | azul | celeste | teal | verde | amarillo
  joinedAt DateTime @default(now()) @map("joined_at")

  team Team @relation(fields: [teamId], references: [id], onDelete: Cascade)
  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([teamId, userId])
  @@unique([teamId, color])
  @@map("team_members")
}

model WorkflowTemplate {
  id        String   @id @default(uuid())
  name      String
  stages    String   // JSON array: ["translator","reviewer","proofreader"]
  ownerId   String?  @map("owner_id")
  isDefault Boolean  @default(false) @map("is_default")
  createdAt DateTime @default(now()) @map("created_at")

  owner    User?     @relation("WorkflowTemplateOwner", fields: [ownerId], references: [id])
  projects Project[] @relation("ProjectWorkflow")

  @@map("workflow_templates")
}

model SegmentAssignment {
  id         String   @id @default(uuid())
  projectId  String   @map("project_id")
  userId     String   @map("user_id")
  rangeStart Int?     @map("range_start")
  rangeEnd   Int?     @map("range_end")
  createdAt  DateTime @default(now()) @map("created_at")

  project Project @relation(fields: [projectId], references: [id], onDelete: Cascade)
  user    User    @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([projectId, userId])
  @@map("segment_assignments")
}

model UserPresence {
  projectId              String   @map("project_id")
  userId                 String   @map("user_id")
  currentSegmentPosition Int      @map("current_segment_position")
  lastHeartbeat          DateTime @default(now()) @map("last_heartbeat")

  project Project @relation(fields: [projectId], references: [id], onDelete: Cascade)
  user    User    @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@id([projectId, userId])
  @@map("user_presence")
}

```typescript
import { NextRequest, NextResponse } from "next/server";
import { getAuthenticatedUser } from "@/lib/api-auth";
import { prisma } from "@/lib/prisma";
import { z } from "zod";

const createSchema = z.object({
  name: z.string().min(1).max(100),
  description: z.string().max(500).optional(),
});

export async function POST(req: NextRequest) {
  const { user, error } = await getAuthenticatedUser();
  if (error || !user) return error ?? NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const body = await req.json();
  const parsed = createSchema.safeParse(body);
  if (!parsed.success) return NextResponse.json({ error: parsed.error.errors[0].message }, { status: 400 });

  const team = await prisma.team.create({
    data: {
      name: parsed.data.name,
      description: parsed.data.description,
      ownerId: user.id,
      members: {
        create: { userId: user.id, role: "pm", color: "azul" },
      },
    },
    include: { members: { include: { user: { select: { id: true, name: true, username: true, avatarUrl: true } } } } },
  });

  return NextResponse.json(team, { status: 201 });
}

export async function GET() {
  const { user, error } = await getAuthenticatedUser();
  if (error || !user) return error ?? NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const teams = await prisma.team.findMany({
    where: {
      OR: [
        { ownerId: user.id },
        { members: { some: { userId: user.id } } },
      ],
    },
    include: {
      members: { include: { user: { select: { id: true, name: true, username: true, avatarUrl: true } } } },
      _count: { select: { projects: true } },
    },
    orderBy: { createdAt: "desc" },
  });

  return NextResponse.json(teams);
}
```

- Create: `src/app/api/segments/[id]/confirm/route.ts`

- [ ] **Step 1: Create PATCH /api/segments/[id]/confirm**

Logic:
1. Auth + verify user has confirmSegments permission
2. Verify segment is in user's assigned range (or user has no range restriction)
3. Get project's workflow template stages
4. Get project's pipelineCheckpoints
5. Current stage index → next stage in template
6. If segment position is in pipelineCheckpoints → set stage to "awaiting_approval", don't advance
7. Else → set workflowStage to next stage, set confirmedBy/confirmedAt
8. Create notification for next-stage user (e.g., reviewer gets "segments_ready")
9. Return updated segment

```typescript
import { NextRequest, NextResponse } from "next/server";
import { getAuthenticatedUser } from "@/lib/api-auth";
import { prisma } from "@/lib/prisma";

export async function PATCH(
  req: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  const { user, error } = await getAuthenticatedUser();
  if (error || !user) return error ?? NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const { id } = await params;

  const segment = await prisma.segment.findUnique({
    where: { id },
    include: {
      project: {
        include: {
          workflowTemplate: true,
          segmentAssignments: true,
          members: true,
        },
      },
    },
  });

  if (!segment) return NextResponse.json({ error: "Segment not found" }, { status: 404 });

  // Verify user is a member
  const membership = segment.project.members.find((m) => m.userId === user.id);
  if (!membership) return NextResponse.json({ error: "Not a member" }, { status: 403 });

  // Verify user's assignment covers this segment
  const assignment = segment.project.segmentAssignments.find((a) => a.userId === user.id);
  if (assignment && assignment.rangeStart != null && assignment.rangeEnd != null) {
    if (segment.position < assignment.rangeStart || segment.position > assignment.rangeEnd) {
      return NextResponse.json({ error: "Segment not in your assigned range" }, { status: 403 });
    }
  }

  // Determine next stage
  const template = segment.project.workflowTemplate;
  const stages: string[] = template ? JSON.parse(template.stages) : ["translator", "reviewer"];
  const stageMap: Record<string, string> = {
    translator: "translating",
    reviewer: "reviewing",
    proofreader: "proofreading",
    dtp: "completing",
  };
  const reverseMap: Record<string, string> = Object.fromEntries(
    Object.entries(stageMap).map(([k, v]) => [v, k])
  );

  const currentRole = reverseMap[segment.workflowStage] || stages[0];
  const currentIdx = stages.indexOf(currentRole);
  const nextIdx = currentIdx + 1;

  let nextStage: string;
  if (nextIdx >= stages.length) {
    nextStage = "completed";
  } else {
    nextStage = stageMap[stages[nextIdx]] || "completed";
  }

  // Check pipeline checkpoints
  const checkpoints: number[] = JSON.parse(segment.project.pipelineCheckpoints || "[]");
  if (checkpoints.includes(segment.position)) {
    nextStage = "awaiting_approval";
  }

  const updated = await prisma.segment.update({
    where: { id },
    data: {
      workflowStage: nextStage,
      confirmedBy: user.id,
      confirmedAt: new Date(),
      status: "confirmed",
    },
  });

  return NextResponse.json(updated);
}
```

### Task 10: Segment reject + error propagation

**Files:**
- Create: `src/app/api/segments/[id]/reject/route.ts`

- [ ] **Step 1: Create PATCH /api/segments/[id]/reject**

Logic:
1. Auth + verify user has rejectSegments permission (reviewer/proofreader)
2. Set segment.workflowStage back to "translating"
3. Set segment.needsRecheck = false (this one is being actively fixed)
4. Find downstream segments from same translator in same section (until paragraph break or range end) → set needsRecheck = true
5. Create notification for translator: "correction_needed" with segment position
6. Return updated segment

```typescript
export async function PATCH(
  req: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  const { user, error } = await getAuthenticatedUser();
  if (error || !user) return error ?? NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const { id } = await params;
  const body = await req.json();
  const reason = body.reason || "";

  const segment = await prisma.segment.findUnique({
    where: { id },
    include: {
      project: {
        include: { segmentAssignments: true, members: true },
      },
    },
  });

  if (!segment) return NextResponse.json({ error: "Segment not found" }, { status: 404 });

  // Verify reviewer membership
  const membership = segment.project.members.find((m) => m.userId === user.id);
  if (!membership) return NextResponse.json({ error: "Not a member" }, { status: 403 });

  // Reset this segment to translating
  const updated = await prisma.segment.update({
    where: { id },
    data: {
      workflowStage: "translating",
      reviewStatus: "rejected",
      reviewedBy: user.id,
      reviewedAt: new Date(),
      needsRecheck: false,
    },
  });

  // Propagate: mark downstream segments from same translator as needsRecheck
  if (segment.confirmedBy) {
    const assignment = segment.project.segmentAssignments.find(
      (a) => a.userId === segment.confirmedBy
    );
    const rangeEnd = assignment?.rangeEnd ?? 999999;

    await prisma.segment.updateMany({
      where: {
        projectId: segment.projectId,
        position: { gt: segment.position, lte: Math.min(segment.position + 20, rangeEnd) },
        confirmedBy: segment.confirmedBy,
        workflowStage: { not: "translating" },
      },
      data: { needsRecheck: true },
    });
  }

  // Create notification for translator
  if (segment.confirmedBy) {
    await prisma.notification.create({
      data: {
        userId: segment.confirmedBy,
        type: "correction_needed",
        title: `SOURCE #${segment.position} was rejected`,
        message: reason || "The reviewer found an issue with this segment.",
        link: `/app/projects/${segment.projectId}?seg=${segment.position}`,
      },
    });
  }

  return NextResponse.json(updated);
}

// POST — heartbeat
export async function POST(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  const { user, error } = await getAuthenticatedUser();
  if (error || !user) return error ?? NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const { id: projectId } = await params;
  const { segmentPosition } = await req.json();

  await prisma.userPresence.upsert({
    where: { projectId_userId: { projectId, userId: user.id } },
    update: { currentSegmentPosition: segmentPosition, lastHeartbeat: new Date() },
    create: { projectId, userId: user.id, currentSegmentPosition: segmentPosition, lastHeartbeat: new Date() },
  });

  return NextResponse.json({ ok: true });
}

// GET — all active presences
export async function GET(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  const { user, error } = await getAuthenticatedUser();
  if (error || !user) return error ?? NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const { id: projectId } = await params;
  const thirtySecondsAgo = new Date(Date.now() - 30_000);

  const presences = await prisma.userPresence.findMany({
    where: { projectId, lastHeartbeat: { gte: thirtySecondsAgo } },
    include: {
      user: { select: { id: true, name: true, username: true, avatarUrl: true } },
    },
  });

  // Get colors from project members
  const members = await prisma.projectMember.findMany({
    where: { projectId },
    select: { userId: true, color: true, role: true },
  });

  const colorMap = Object.fromEntries(members.map((m) => [m.userId, m.color]));

  const result = presences.map((p) => ({
    ...p,
    color: colorMap[p.userId] || "azul",
  }));

  return NextResponse.json(result);
}
```
